The Arrange-Act-Assert (AAA) Pattern Link to heading

The most widely used pattern for structuring unit tests.

Structure Link to heading

def test_user_registration():
    # Arrange
    user_data = {
        'username': 'testuser',
        'email': 'test@example.com',
        'password': 'SecurePass123!'
    }
    db = Database()
    
    # Act
    result = register_user(user_data, db)
    
    # Assert
    assert result.success is True
    assert result.user_id is not None
    assert db.user_exists('testuser')

Why AAA Works Link to heading

  • Clear separation - Each phase has a specific purpose
  • Readable - Anyone can understand the test flow
  • Maintainable - Easy to identify which part needs updating

Best Practices for AAA Link to heading

  1. Keep arrange short - Use test data factories
  2. One action per test - Don’t test multiple operations
  3. Comprehensive assertions - Test both positive and negative cases

The Given-When-Then (GWT) Pattern Link to heading

Popular in Behavior-Driven Development (BDD), great for acceptance tests.

Structure Link to heading

Feature: User Login

  Scenario: Successful login with valid credentials
    Given the user is on the login page
    And the user has valid credentials
    When the user enters correct username and password
    And clicks the login button
    Then the user should be redirected to the dashboard
    And see a welcome message

Implementing GWT in Code Link to heading

// JavaScript/Playwright example
test('successful login', async ({ page }) => {
  // Given
  await page.goto('/login');
  
  // When
  await page.fill('[data-testid="username"]', 'testuser');
  await page.fill('[data-testid="password"]', 'password123');
  await page.click('[data-testid="login-button"]');
  
  // Then
  await expect(page).toHaveURL('/dashboard');
  await expect(page.locator('.welcome-message')).toBeVisible();
});

When to Use GWT Link to heading

  • Acceptance testing
  • Integration tests
  • Feature validation
  • Communication with stakeholders

The Factory Pattern Link to heading

Creating test data efficiently using factories.

Simple Factory Link to heading

class UserFactory:
    @staticmethod
    def create(**kwargs):
        defaults = {
            'username': 'testuser',
            'email': 'test@example.com',
            'password': 'SecurePass123!',
            'active': True
        }
        defaults.update(kwargs)
        return User(**defaults)
    
    @staticmethod
    def create_batch(count, **kwargs):
        return [UserFactory.create(**kwargs) for _ in range(count)]

Using the Factory Link to heading

# Create a single user
user = UserFactory.create()

# Create a user with custom data
admin = UserFactory.create(username='admin', role='admin')

# Create multiple users
users = UserFactory.create_batch(10)

Benefits Link to heading

  • Consistency - Same data structure across tests
  • Flexibility - Override defaults easily
  • Reusability - Share across test files

The Builder Pattern Link to heading

Construct complex test objects step by step.

Implementation Link to heading

public class UserBuilder {
    private String username = "defaultuser";
    private String email = "default@example.com";
    private List<String> roles = new ArrayList<>();
    private boolean active = true;
    
    public UserBuilder withUsername(String username) {
        this.username = username;
        return this;
    }
    
    public UserBuilder withEmail(String email) {
        this.email = email;
        return this;
    }
    
    public UserBuilder withRole(String role) {
        this.roles.add(role);
        return this;
    }
    
    public UserBuilder inactive() {
        this.active = false;
        return this;
    }
    
    public User build() {
        return new User(username, email, roles, active);
    }
}

Using the Builder Link to heading

// Simple user
User user = new UserBuilder().build();

// Admin user with custom email
User admin = new UserBuilder()
    .withUsername("admin")
    .withEmail("admin@company.com")
    .withRole("admin")
    .withRole("moderator")
    .build();

// Inactive user
User inactiveUser = new UserBuilder()
    .withUsername("inactive")
    .inactive()
    .build();

Parameterized Tests Link to heading

Run the same test with different inputs.

Python Example Link to heading

import pytest

@pytest.mark.parametrize("input,expected", [
    (2, 4),
    (3, 9),
    (5, 25),
    (10, 100),
])
def test_square(input, expected):
    assert input ** 2 == expected

Java Example Link to heading

@ParameterizedTest
@CsvSource({
    "admin, admin@example.com, true",
    "user, user@example.com, false",
    "guest, guest@example.com, false"
})
void testUserCreation(String username, String email, boolean isAdmin) {
    User user = userService.create(username, email);
    
    assertEquals(username, user.getUsername());
    assertEquals(email, user.getEmail());
    assertEquals(isAdmin, user.isAdmin());
}

Playwright Example Link to heading

const testCases = [
  { username: 'admin', password: 'admin123', expected: 'dashboard' },

  });
});

Page Object Model (POM) Link to heading

A design pattern for web UI automation.

Page Object Link to heading

class LoginPage:
    def __init__(self, page):
        self.page = page
    
    @property
    def username_input(self):
        return self.page.locator('[data-testid="username"]')
    
    @property
    def password_input(self):
        return self.page.locator('[data-testid="password"]')
    
    @property
    def login_button(self):
        return self.page.locator('[data-testid="login-button"]')
    
    @property
    def error_message(self):
        return self.page.locator('.error-message')
    
    def login(self, username, password):
        self.username_input.fill(username);
        self.password_input.fill(password);
        self.login_button.click();
    
    def login_as_admin(self):
        self.login('admin', 'admin123')

Using the Page Object Link to heading

def test_login_success(page):
    login_page = LoginPage(page)
    
    login_page.goto()
    login_page.login_as_admin()
    
    dashboard = DashboardPage(page)
    expect(dashboard.welcome_message).to_be_visible()

Test Data Cleanup Patterns Link to heading

Proper cleanup prevents test pollution.

Fixture-Based Cleanup Link to heading

import pytest
from database import Database

@pytest.fixture
def clean_database():
    db = Database()
    yield db
    # Cleanup after test
    db.cleanup_all()

def test_user_creation(clean_database):
    user = clean_database.create_user('test')
    assert clean_database.user_exists('test')

Context Manager Cleanup Link to heading

class TemporaryFile:
    def __init__(self):
        self.path = create_temp_file()
    
    def __enter__(self):
        return self.path
    
    def __exit__(self, *args):
        delete_file(self.path)

def test_file_processing():
    with TemporaryFile() as filepath:
        result = process_file(filepath)
        assert result.success

Anti-Patterns to Avoid Link to heading

1. Test Interdependence Link to heading

# BAD - Tests depend on each other
def test_create_user():
    user_id = create_user('test')
    assert user_id == 1  # Assumes previous test ran

# GOOD - Each test is independent
def test_create_user():
    user_id = create_user('unique_test_user_' + str(uuid.uuid4()))
    assert user_id is not None

2. Magic Numbers Link to heading

# BAD
assert response_time < 5000

# GOOD
MAX_RESPONSE_TIME_MS = 5000
assert response_time < MAX_RESPONSE_TIME_MS

3. Testing Too Much Link to heading

# BAD - Multiple assertions, multiple actions
def test_everything():
    user = create_user()
    login(user)
    create_project()
    add_task()
    assert everything works

# GOOD - One action per test
def test_user_creation():
    user = create_user()
    assert user.id is not None

def test_user_login():
    user = create_user()
    session = login(user)
    assert session.is_authenticated
